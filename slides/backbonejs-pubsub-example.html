Backbone.js Pubsub Example

And it functions something like:

var ViewOne = Backbone.View.extend({
  fnOne: function() {
    this.pubSub.trigger('event:name', {});
  }
});
var ViewTwo = Backbone.View.extend({
  initialize: function() {
    this.listenTo(this.pubSub, 'event:name', doSomething);
  },
  doSomething: function(data) {
    // do something with data
  }
});
