Backbone.js Model REST Usage

var AddressModel Backbone.Model.extend({
  idAttribute: 'addressId',
  urlRoot: '/themes/build-tablet/api/account/address',
  getType: function() {
    return this.get('addressTypeId') === 2 ?
      'shipping' :
      'billing';
  }

});
var view = new Backbone.View.extend({
  model: new AddressModel({
    // address data
  }),
});
view.model.fetch();
